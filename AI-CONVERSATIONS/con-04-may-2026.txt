def savings_goal_calculator(current_savings, monthly_contribution, target_amount):
    """
    Calculates how many months are needed to reach a savings target.
    
    Args:
        current_savings: Amount already saved (AUD)
        monthly_contribution: Amount saved per month (AUD)
        target_amount: Savings goal (AUD)
    
    Returns:
        str: Human-readable result message
    """
    try:
        current_savings = float(current_savings)
        monthly_contribution = float(monthly_contribution)
        target_amount = float(target_amount)
        
        if target_amount <= current_savings:
            return "🎉 You have already reached your savings goal!"
        
        if monthly_contribution <= 0:
            return "❌ Monthly contribution must be greater than zero."
        
        remaining = target_amount - current_savings
        months = remaining / monthly_contribution
        
        years = int(months // 12)
        extra_months = round(months % 12, 1)
        
        if years > 0:
            return f"You need approximately {years} year(s) and {extra_months} month(s) to reach your goal of AUD {target_amount:.2f}."
        else:
            return f"You need approximately {months:.1f} months to reach your goal of AUD {target_amount:.2f}."
    
    except (ValueError, TypeError):
        return "❌ Invalid input. Please enter numeric values for all fields."
